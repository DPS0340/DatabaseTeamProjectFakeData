CREATE TABLE member (
memberNumber integer NOT NULL PRIMARY KEY,
password char(16),
name char(32),
englishName char(64),
birthDate char(6),
memberRating char(32),
callNumber char(11),
phoneNumber char(11),
address char(128),
zipCode char(10),
email char(32));

CREATE TABLE businessMember (
businessNumber integer NOT NULL PRIMARY KEY,
memberNumber integer FOREIGN KEY REFERENCES number(businessMember),
koreanCorpName char(64),
englishCorpName char(64),
corpCallNumber char(11),
corpAddress char(64),
corpZipCode char(10));

CREATE TABLE memberCard (
cardCode integer NOT NULL PRIMARY KEY,
memberNumber integer FOREIGN KEY REFERENCES number(memberCard),
cardName char(32),
cardBenefit char(128),
ratingPrice integer,
boughtDate char(120),
expiredDate char(120));

CREATE TABLE anorder (
orderNumber integer NOT NULL PRIMARY KEY,
memberNumber integer FOREIGN KEY REFERENCES number(anorder),
productNumber integer,
deliveryCorpNumber integer,
totalPrice integer,
totalQuantity integer,
boughtTime char(120));

CREATE TABLE product (
productCode integer NOT NULL PRIMARY KEY,
categorynumber integer,
productName char(32),
productCorp char(32),
productImage char(120),
deliveryAndRefund char(20),
reviewContent char(128),
productSpec char(128),
productPrice integer,
categoryCode integer FOREIGN KEY REFERENCES code(product));

CREATE TABLE category (
categoryCode integer NOT NULL PRIMARY KEY,
categoryName char(32));

CREATE TABLE review (
postNumber integer NOT NULL PRIMARY KEY,
productNumber integer,
memberNumber integer FOREIGN KEY REFERENCES number(review),
categoryCode integer FOREIGN KEY REFERENCES code(review),
writeDate char(120),
content char(128),
likes integer);

CREATE TABLE include (
orderNumber undefined NOT NULL PRIMARY KEY,
memberNumber integer FOREIGN KEY REFERENCES number(include),
productCode integer FOREIGN KEY REFERENCES code(include),
categoryCode integer FOREIGN KEY REFERENCES code(include),
quantity integer,
sumPrice integer);

CREATE TABLE delivery (
deliveryNumber integer NOT NULL PRIMARY KEY,
deliveryStart char(32),
deliveryDestination char(32),
deliveryDate char(120));

CREATE TABLE send (
deliveryNumber integer NOT NULL PRIMARY KEY,
productCode integer FOREIGN KEY REFERENCES code(send),
categoryCode integer FOREIGN KEY REFERENCES code(send),
totalTime integer);

CREATE TABLE refresh (
productCode integer NOT NULL PRIMARY KEY,
categoryCode integer FOREIGN KEY REFERENCES code(refresh),
remainsNumber integer FOREIGN KEY REFERENCES number(refresh),
refreshDate char(120));

CREATE TABLE remains (
remainsNumber integer NOT NULL PRIMARY KEY,
remainsQuantity integer,
remainsDate char(120));

CREATE TABLE checked (
employeeNumber integer NOT NULL PRIMARY KEY,
remainsNumber integer FOREIGN KEY REFERENCES number(checked),
checkedDate char(120),
others char(20));

CREATE TABLE employee (
employeeNumber integer NOT NULL PRIMARY KEY,
employeeName char(32),
employeeSSN integer,
employeePhoneNumber char(32),
employeeEmail char(32),
employeeWorkedDate char(120),
employeeSalary integer);

CREATE TABLE request (
orderNumber integer NOT NULL PRIMARY KEY,
memberNumber integer FOREIGN KEY REFERENCES number(request),
deliveryNumber integer FOREIGN KEY REFERENCES number(request),
requestedDate char(120));
